Longest Subarray with given Sum K(Positives)
public class Solution {
    public static int longestSubarrayWithSumK(int []a, long k) {
        int left=0;
        int right=0;
        int sum=a[0];
        int cons=0;
        int n=a.length;
        while(right<n)
        {
            while(left<=right && sum>k)
            {
                sum-=a[left];
                left++;
            }
            if(sum==k)
            {
                cons=Math.max(cons,right-left+1);
                
            }
            right++;
            if(right<n)
            {
            sum+=a[right];
            }
        }
        return cons;
    }
}
